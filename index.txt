    <script type="module" id="coreLogic">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInWithCustomToken, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, doc, setDoc, getDocs, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // --- GESTIONE ERRORI FIREBASE OFFLINE/LOCALE ---
        let app, auth, db;
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        let currentUser = null;

        try {
            const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
            if (firebaseConfig && Object.keys(firebaseConfig).length > 0) {
                app = initializeApp(firebaseConfig);
                auth = getAuth(app);
                db = getFirestore(app);
            } else {
                console.warn("Attenzione: Configurazione Firebase Cloud non rilevata. L'App funzionerà in modalità 'Solo Locale'.");
            }
        } catch (e) {
            console.error("Errore inizializzazione Firebase:", e);
        }

        // Esponi le funzioni globalmente per l'HTML
        window.saveTree = saveTree;
        window.showDeleteModal = showDeleteModal;
        window.closeDeleteModal = closeDeleteModal;
        window.closeErrorModal = closeErrorModal;
        window.exportCSV = exportCSV;
        window.exportPDF = exportPDF;
        window.exportZIP = exportZIP;
        window.handlePhoto = handlePhoto;
        window.forceCloudSync = forceCloudSync;
        window.initMap = initMap;
        window.placeMarkerAndGetUTM = placeMarkerAndGetUTM;
        window.retryGpsConnection = retryGpsConnection;
        window.closeGpsPermissionModal = closeGpsPermissionModal;
        window.saveApiKeyAndLoadMap = saveApiKeyAndLoadMap;

        // --- STATO DELL'APP ---
        let map;
        let userMarker; // Posizione GPS dell'utente
        let pinMarker;  // Posizione toccata per la pianta
        let treesDatabase = []; // Array per memorizzare i record
        let treeToDeleteId = null; // Memorizza l'ID in attesa di cancellazione

        // --- AUTENTICAZIONE E CLOUD LOAD ---
        async function initAuth() {
            if (!auth) return; // Se Firebase non è configurato, salta
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (error) {
                console.error("Errore Autenticazione:", error);
            }
        }

        if (auth) {
            onAuthStateChanged(auth, (user) => {
                currentUser = user;
                if (user) {
                    loadFromCloud();
                }
            });
        }

        async function loadFromCloud() {
            if (!currentUser || !db) return;
            try {
                showToast("Caricamento dati Cloud in corso...");
                const rilieviRef = collection(db, 'artifacts', appId, 'users', currentUser.uid, 'rilievi_olivi');
                const snapshot = await getDocs(rilieviRef);
                if (!snapshot.empty) {
                    treesDatabase = [];
                    snapshot.forEach(doc => {
                        treesDatabase.push(doc.data());
                    });
                    
                    updateNextId();
                    document.getElementById("recordCount").innerText = treesDatabase.length;
                    renderTreeList();
                    showToast("Dati Cloud recuperati con successo.");
                }
            } catch (e) {
                console.error("Errore lettura Cloud:", e);
                showToast("Impossibile caricare dal Cloud.");
            }
        }

        async function saveTreeToCloud(treeRecord) {
            if (!currentUser || !db) return false;
            try {
                const docRef = doc(db, 'artifacts', appId, 'users', currentUser.uid, 'rilievi_olivi', treeRecord.id);
                await setDoc(docRef, treeRecord);
                return true;
            } catch (e) {
                console.error("Errore scrittura Cloud:", e);
                return false;
            }
        }

        async function forceCloudSync() {
            if (!currentUser || !db) {
                showErrorModal("Database Cloud non configurato in questo ambiente. Sincronizzazione fallita.");
                return;
            }
            if (treesDatabase.length === 0) {
                showToast("Nessun dato da sincronizzare.");
                return;
            }

            showToast("Sincronizzazione di massa in corso...");
            let count = 0;
            for (const tree of treesDatabase) {
                const success = await saveTreeToCloud(tree);
                if(success) count++;
            }
            showToast(`Sincronizzazione completata: ${count}/${treesDatabase.length} record sul Cloud.`);
        }

        // --- INIZIALIZZAZIONE GOOGLE MAPS DINAMICA E API KEY ---
        function loadGoogleMapsScript() {
            const savedKey = localStorage.getItem('gmaps_api_key');
            if (savedKey) {
                injectGoogleMapsScript(savedKey);
            } else {
                document.getElementById("apiKeyModal").classList.remove("hidden");
                document.getElementById("gpsStatus").innerText = "In attesa di API Key...";
            }
        }

        function saveApiKeyAndLoadMap() {
            const keyInput = document.getElementById("apiKeyValue").value.trim();
            if (!keyInput) {
                showErrorModal("Inserisci una API Key valida per continuare.");
                return;
            }
            localStorage.setItem('gmaps_api_key', keyInput);
            document.getElementById("apiKeyModal").classList.add("hidden");
            injectGoogleMapsScript(keyInput);
        }

        function injectGoogleMapsScript(apiKey) {
            document.getElementById("gpsStatus").innerText = "Caricamento librerie mappa...";

            const script = document.createElement('script');
            script.src = `https://maps.googleapis.com/maps/api/js?key=${apiKey}&callback=initMap`;
            script.async = true;
            script.defer = true;
            
            // Gestione errore di rete nel caricamento dello script
            script.onerror = function() {
                document.getElementById("gpsStatus").innerText = "Impossibile caricare la cartografia.";
            };

            document.head.appendChild(script);
        }
        
        // Gestione globale dell'errore di autenticazione Google Maps (es. chiave errata o revocata)
        window.gm_authFailure = function() {
            showErrorModal("L'API Key di Google Maps inserita non è valida o ha restrizioni errate. Reinseriscila.");
            localStorage.removeItem('gmaps_api_key');
            document.getElementById("gpsStatus").innerText = "Errore API Key.";
            setTimeout(() => {
                document.getElementById("apiKeyValue").value = "";
                document.getElementById("apiKeyModal").classList.remove("hidden");
            }, 2500);
        };

        // --- INIZIALIZZAZIONE MAPPA E GPS ---
        function initMap() {
            document.getElementById("gpsStatus").innerText = "Ricerca GPS in corso...";
            // Coordinate di default: Noci, Puglia (40.793, 17.126)
            const defaultLoc = { lat: 40.793, lng: 17.126 };

            map = new google.maps.Map(document.getElementById("map"), {
                zoom: 18,
                center: defaultLoc,
                mapTypeId: "satellite", // Ortofoto aggiornata
                disableDefaultUI: true, // UI pulita per mobile
                zoomControl: true
            });

            // Listener per il tocco sulla mappa
            map.addListener("click", (e) => {
                placeMarkerAndGetUTM(e.latLng);
            });

            // Avvia la geolocalizzazione GPS in real-time
            startGPS();
        }

        function startGPS() {
            const statusText = document.getElementById("gpsStatus");
            if (navigator.geolocation) {
                navigator.geolocation.watchPosition(
                    (position) => {
                        const pos = {
                            lat: position.coords.latitude,
                            lng: position.coords.longitude,
                        };

                        statusText.innerText = "GPS Attivo - Precisione: " + Math.round(position.coords.accuracy) + "m";
                        statusText.classList.replace("text-gray-300", "text-green-400");

                        // Centra la mappa solo la prima volta o se richiesto (qui aggiorniamo solo il marker)
                        if (!userMarker) {
                            userMarker = new google.maps.Marker({
                                position: pos,
                                map: map,
                                icon: {

                            });

                        } else {

                        }

                    (error) => {

                        }
                        
                        let msg = "Errore GPS.";
                        if (error.code === 1) {
                            // Se siamo nell'ambiente di anteprima isolato (iframe)
                            if (error.message && error.message.includes("permissions policy")) {
                                msg = "Anteprima: GPS bloccato, uso posizione simulata (Noci).";
                                const mockPos = { lat: 40.793, lng: 17.126 };

                                            strokeColor: "white",
                                            strokeWeight: 2,

                                        title: "Posizione Simulata (Noci)"
                                    });
                                    if (map) map.setCenter(mockPos);
                                } else {
                                    userMarker.setPosition(mockPos);
                                }
                            } else {
                                msg = "GPS negato (permessi o ambiente non sicuro).";
                                // Se l'errore è dovuto alla policy del browser, mostriamo la modale di guida
                                showGpsPermissionModal();
                            }
                        }
                        else if (error.code === 2) msg = "Segnale GPS non disponibile.";
                        else if (error.code === 3) msg = "Timeout ricerca GPS.";
                        
                        statusText.innerText = msg;
                        statusText.classList.replace("text-gray-300", "text-red-400");
                        if (msg.includes("Simulata")) {
                            statusText.classList.replace("text-red-400", "text-yellow-400");
                        }
                    },
                    { enableHighAccuracy: true, maximumAge: 0, timeout: 10000 }
                );
            } else {
                statusText.innerText = "GPS non supportato da questo browser (o serve HTTPS).";
            }
        }

        // --- CONVERSIONE COORDINATE (WGS84 -> UTM) ---
        function placeMarkerAndGetUTM(latLng) {
            // Posiziona o sposta il pin della pianta
            if (!pinMarker) {
                pinMarker = new google.maps.Marker({
                    position: latLng,
                    map: map,
                    icon: "http://maps.google.com/mapfiles/ms/icons/red-dot.png"
                });
            } else {
                pinMarker.setPosition(latLng);
            }

            const lat = latLng.lat();
            const lng = latLng.lng();

            // Calcolo zona UTM dinamica in base alla longitudine
            // (La Puglia è generalmente Zona 33N)
            const zone = Math.floor((lng + 180) / 6) + 1;
            
            // Definizioni per Proj4js
            const projWGS84 = "+proj=longlat +datum=WGS84 +no_defs";
            const projUTM = `+proj=utm +zone=${zone} +datum=WGS84 +units=m +no_defs`;

            // Esecuzione conversione
            const utmCoords = proj4(projWGS84, projUTM, [lng, lat]);
            
            // Regola di formattazione: Punto per i decimali, Nessun separatore migliaia, Suffisso " m E" / " m N"
            // utmCoords[0] è Easting (X), utmCoords[1] è Northing (Y)
            const easting = utmCoords[0].toFixed(2);
            const northing = utmCoords[1].toFixed(2);

            document.getElementById("coordE").value = `${easting} m E`;
            document.getElementById("coordN").value = `${northing} m N`;
        }

        // --- GESTIONE FOTOGRAFIA (CON COMPRESSIONE AUTOMATICA) ---
        function handlePhoto(event) {
            const file = event.target.files[0];
            if (!file) return;

            // Generazione nome file aaaammgg_hhmmss.jpg
            const now = new Date();
            const pad = (n) => String(n).padStart(2, '0');
            const fileName = `${now.getFullYear()}${pad(now.getMonth()+1)}${pad(now.getDate())}_${pad(now.getHours())}${pad(now.getMinutes())}${pad(now.getSeconds())}.jpg`;
            
            document.getElementById("photoFileName").value = fileName;
            document.getElementById("photoNameDisplay").innerText = "Elaborazione foto in corso...";

            // Lettura file con FileReader
            const reader = new FileReader();
            reader.onload = function(e) {
                // Compressione con Canvas per rispettare il limite Cloud (1MB max) ed evitare file pesanti
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    // Aumentata la risoluzione nativa a 1920x1440 (Full HD 4:3) per l'export in alta qualità
                    const MAX_WIDTH = 1920; 
                    const MAX_HEIGHT = 1440;
                    let width = img.width;
                    let height = img.height;

                    if (width > height) {
                        if (width > MAX_WIDTH) {
                            height *= MAX_WIDTH / width;
                            width = MAX_WIDTH;
                        }
                    } else {
                        if (height > MAX_HEIGHT) {
                            width *= MAX_HEIGHT / height;
                            height = MAX_HEIGHT;
                        }
                    }
                    canvas.width = width;
                    canvas.height = height;
                    const ctx = canvas.getContext('2d');
                    ctx.drawImage(img, 0, 0, width, height);
                    
                    // Comprime a JPEG qualità 85% (Ottimo bilanciamento tra definizione e limite Cloud di 1MB)
                    const compressedDataUrl = canvas.toDataURL('image/jpeg', 0.85); 

                    const preview = document.getElementById("photoPreview");
                    preview.src = compressedDataUrl;
                    preview.style.display = "block";
                    document.getElementById("photoDataUrl").value = compressedDataUrl;
                    document.getElementById("photoNameDisplay").innerText = "File: " + fileName;
                }
                img.src = e.target.result;
            }
            reader.readAsDataURL(file);
        }

                return;

                coord_n: coordN,

        // --- GESTIONE RECORD, ID E CANCELLAZIONI ---
        function updateNextId() {
            const usedIds = treesDatabase.map(t => parseInt(t.id, 10)).sort((a,b) => a - b);
            let nextId = 1;
            // Algoritmo "Fill the gap": Cerca il primo ID mancante partendo da 1
            for (let i = 0; i < usedIds.length; i++) {
                if (usedIds[i] === nextId) {
                    nextId++;
                } else {
                    break; // Trovato un ID saltato/cancellato
                }
            }
            document.getElementById("idPianta").value = String(nextId).padStart(3, '0');
        }

        function exportPDF() {
            if (treesDatabase.length === 0) {
                showErrorModal("Nessun dato da esportare.");
                return;
            }

            const { jsPDF } = window.jspdf;
            const doc = new jsPDF('landscape'); // Formato orizzontale per far entrare le tabelle

            // Intestazione Documento
            doc.setFontSize(15);
            doc.text("Classificazione olivi ai sensi dell’Art. 2 della Legge Regione Puglia 4 giugno 2007, n. 14", 14, 20);
            doc.setFontSize(11);
            const dateStr = new Date().toLocaleDateString('it-IT');
            doc.text(`Data Rilievo: ${dateStr} - Totale Piante: ${treesDatabase.length}`, 14, 28);
            
            // Tabella Riepilogativa tramite autoTable
            const tableColumn = ["ID", "Comune", "Fg.", "Part.", "Nord (UTM)", "Est (UTM)", "Classe Diametrica", "Stato", "Ambiente"];
            const tableRows = [];

            treesDatabase.forEach(tree => {
                const treeData = [
                    tree.id,
                    tree.comune,
                    tree.foglio,
                    tree.particella,
                    tree.coord_n,
                    tree.coord_e,
                    tree.classe_diametrica,
                    tree.monumentalita,
                    tree.ambiente
                ];
                tableRows.push(treeData);
            });

            doc.autoTable({
                head: [tableColumn],
                body: tableRows,
                startY: 35,
                styles: { fontSize: 8, cellPadding: 2 },
                headStyles: { fillColor: [22, 163, 74] } // Verde Agronomo
            });

            // Opzionale: Aggiunta schede fotografiche se presenti
            // Generiamo schede con 2 foto per pagina (affiancate e centrate)
            const treesWithPhotos = treesDatabase.filter(t => t.foto_dati);
            
            for (let i = 0; i < treesWithPhotos.length; i++) {
                // Aggiungi una nuova pagina ogni 2 foto
                if (i % 2 === 0) {
                    doc.addPage();
                }
                
                const tree = treesWithPhotos[i];
                const isLeft = (i % 2 === 0); // true per la foto a sinistra, false per quella a destra
                
                // Calcolo asse X centrale per la mezza pagina (Larghezza A4 orizzontale = 297mm)
                // Metà sinistra: 148.5mm -> Centro = 74.25mm
                // Metà destra: da 148.5 a 297 -> Centro = 222.75mm
                const centerX = isLeft ? 74.25 : 222.75;
                
                // Testi Intestazione Foto (Centrati verticalmente e orizzontalmente)
                doc.setFontSize(14);
                doc.text(`Scheda Pianta ID: ${tree.id}`, centerX, 20, { align: 'center' });
                
                doc.setFontSize(10);
                doc.text(`Comune: ${tree.comune} - Fg: ${tree.foglio} - Part: ${tree.particella}`, centerX, 28, { align: 'center' });
                doc.text(`Coordinate: ${tree.coord_n} / ${tree.coord_e}`, centerX, 34, { align: 'center' });
                doc.text(`File: ${tree.foto_nome}`, centerX, 40, { align: 'center' });
                
                try {
                    // Previene lo stretching calcolando le proporzioni reali dell'immagine
                    const imgProps = doc.getImageProperties(tree.foto_dati);
                    const pdfMaxWidth = 135; // Spazio orizzontale max per mezza pagina (con margini)
                    const pdfMaxHeight = 150; // Spazio verticale max per la foto
                    
                    // Calcola il rapporto di scala necessario per far rientrare l'immagine nel riquadro massimo
                    const ratio = Math.min(pdfMaxWidth / imgProps.width, pdfMaxHeight / imgProps.height);
                    
                    // Applica il rapporto alle dimensioni reali
                    const finalWidth = imgProps.width * ratio;
                    const finalHeight = imgProps.height * ratio;
                    
                    // Calcolo posizionamento Y e X per centrare l'immagine nel riquadro
                    const imgX = centerX - (finalWidth / 2);
                    const boxCenterY = 45 + (pdfMaxHeight / 2); // Centro dell'area dedicata alla foto
                    const imgY = boxCenterY - (finalHeight / 2);
                    
                    // 1. Disegna l'ombreggiatura (rettangolo grigio chiaro spostato in basso a destra)
                    doc.setFillColor(210, 210, 210); // Grigio ombra
                    doc.rect(imgX + 1.5, imgY + 1.5, finalWidth, finalHeight, 'F');
                    
                    // 2. Inserisce la Fotografia
                    doc.addImage(tree.foto_dati, 'JPEG', imgX, imgY, finalWidth, finalHeight);
                    
                    // 3. Disegna la cornice elegante (sottile bordo scuro)
                    doc.setDrawColor(80, 80, 80); // Grigio scuro/antracite
                    doc.setLineWidth(0.3); // Spessore fine
                    doc.rect(imgX, imgY, finalWidth, finalHeight, 'S');
                    
                } catch(e) {
                    console.log("Errore inserimento immagine per ID", tree.id, e);
                    doc.text("Errore rendering immagine.", centerX, 55, { align: 'center' });
                }
            }

            // --- Aggiunta Numerazione Pagine a Piè di Pagina ---
            const totalPages = doc.internal.getNumberOfPages();
            
            // Recuperiamo il nome del Tecnico dal primo record utile (o dal form se vuoto)
            const nomeTecnico = treesDatabase.length > 0 && treesDatabase[0].tecnico ? treesDatabase[0].tecnico : document.getElementById("tecnico").value;

            for (let i = 1; i <= totalPages; i++) {
                doc.setPage(i);
                
                // Numerazione pagina (centrata)
                doc.setFont("helvetica", "normal");
                doc.setFontSize(10);
                doc.setTextColor(80, 80, 80); // Grigio scuro per il footer
                doc.text(`Pag. ${i} di ${totalPages}`, 148.5, 202, { align: 'center' });
                
                // Riferimenti Tecnico (A destra, font corsivo, size 9)
                doc.setFont("helvetica", "italic");
                doc.setFontSize(9);
                doc.text(`${nomeTecnico}`, 283, 202, { align: 'right' });
            }

            doc.save(`Report_Olivi_${new Date().toISOString().slice(0,10)}.pdf`);
            showToast("Download PDF avviato");
        }

        // --- GESTIONE VERSIONE AUTOMATICA (HASH DEL CODICE) ---
        function setAppVersion() {
            // Estraiamo il contenuto testuale del Form, dello Stile e dello Script logico.
            // Calcolando un "Hash" (Checksum) di questi testi, la versione cambierà matematicamente 
            // SOLO ED ESCLUSIVAMENTE se viene aggiunto, tolto o modificato anche un solo carattere nel codice sorgente.
            const formHTML = document.getElementById("treeForm") ? document.getElementById("treeForm").innerHTML : "";
            const cssApp = document.getElementsByTagName("style")[0] ? document.getElementsByTagName("style")[0].innerHTML : "";
            const scriptLogica = document.getElementById("coreLogic") ? document.getElementById("coreLogic").innerHTML : "";
            
            const codiceSorgente = formHTML + cssApp + scriptLogica;
            
            let hash = 0;
            if (codiceSorgente.length === 0) return;
            for (let i = 0; i < codiceSorgente.length; i++) {
                const char = codiceSorgente.charCodeAt(i);
                hash = ((hash << 5) - hash) + char;
                hash = hash & hash; // Converte in intero a 32 bit
            }
            
            // Genera una stringa esadecimale univoca per la build corrente (es. 7A3F9B)
            const hashVersion = Math.abs(hash).toString(16).toUpperCase().substring(0, 6);
            document.getElementById("appVersion").innerText = "v.Build-" + hashVersion;
        }
